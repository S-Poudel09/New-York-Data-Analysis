from scipy.stats import f_oneway  # Import the ANOVA function from SciPy

# Assume Response_Time is equal to Request_Closing_Time for analysis purposes
df_cleaned['Response_Time'] = df_cleaned['Request_Closing_Time']

# Select relevant columns and drop any rows with missing values
anova_data = df_cleaned[['Complaint Type', 'Response_Time']].dropna()

# Group the data by complaint type and extract the response times for each group
grouped = [group['Response_Time'] for name, group in anova_data.groupby('Complaint Type')]

# Perform one-way ANOVA test to compare average response times across complaint types
f_stat, p_val = f_oneway(*grouped)

# Print the test results
print("Test 1 - ANOVA")
print("F-statistic:", f_stat)  # Measure of variation between group means
print("P-value:", p_val)       # Probability of observing the result if H0 is true

# Interpret the result based on significance level ( = 0.05)
if p_val < 0.05:
    print("Reject H0: Average response times vary across complaint types.")
else:
    print("Fail to reject H0: No significant difference in average response times.")
